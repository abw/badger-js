    Return:

    Promise

    fulfills with the file content

    Example:

    file('myfile.txt').read().then( text => console.log(text) );
    file('myfile.json', { codec: 'json' }).read().( data => console.log(data) );
    file('myfile.json').read({ codec: 'json' }).then( data => console.log(data) );
